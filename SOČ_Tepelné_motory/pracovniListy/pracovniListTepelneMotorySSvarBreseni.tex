    \item \textbf{Vypočítejte práci, kterou vykoná parní stroj, který dostane páru o teplotě 227 °C a vypustí ji o 50 K chladnější. Jeho účinnost je 8 \% a spotřebuje 50 g páry. Měrná tepelná kapacita páry je 1840 J/kg\(\cdot\)K.}

        \begin{minipage}{0.5\textwidth}
            \begin{center}
                \textcolor{red}{
                \begin{tabular}{l l}
                     \(W\) & ? J\\
                     \(\Delta{T}\) & 50 K\\
                     \(\eta\)& 8 \% = 0,08 \\
                     \(m\) & 50g = 0,05 kg\\
                     \(c\) & 1840 \mertepkap 
                \end{tabular}
                }
            \end{center}

        \end{minipage}

            \redeq{Q_d=m\cdot{c}\cdot{\Delta{T}}}
            \redeq{W=(m\cdot{c}\cdot{\Delta{T}})\cdot{\eta}}
            \redeq{W=(0,05\cdot{1840}\cdot{50})\cdot{0,08}}
            \redeq{W=368}
        \end{minipage}

    \item \textbf{Raketa s tahem 45 kN má 5875 kg paliva. Kolik minut poletí, pokud má výstupní rychlost 955 \ms a o 14 kPa vyšší výstupní tlak než prostředí? Poloměr trysky je 20 cm.}

        \begin{minipage}{0.5\textwidth}
            \begin{center}
                \textcolor{red}{
                \begin{tabular}{l l}
                    \(t\) & ? s\\
                    \(\dot{m}\) & ? \kgs\\
                    \(F\) & 45 kN = 45000 N\\
                    \(m\) & 5875 kg\\
                    \(v\) & 955 \ms\\
                    \(\Delta{p}\) & 14 kPa = 14000 Pa\\
                    \(r\) & 20 cm = 0,2 m
                \end{tabular}
                }
            \end{center}
            \redeq{t=\frac{m}{\dot{m}}}
            \redeq{F=\dot{m}\cdot{v}+\Delta{p}\cdot{S}} 
        \end{minipage}
        \begin{minipage}{0.5\textwidth}
            \redeq{S=\pi\cdot{r^2}}
            \redeq{F=\dot{m}\cdot{v}+\Delta{p}\cdot{\pi\cdot{r^2}}}
            \redeq{\dot{m}=\frac{F+\Delta{p}\cdot{\pi\cdot{r^2}}}{v}}
            \redeq{t=\frac{m}{\frac{F+\Delta{p}\cdot{\pi\cdot{r^2}}}{v}}}
            \redeq{t=\frac{5875}{\frac{45000+14000\cdot{\pi\cdot{0,2^2}}}{955}}}
            \redeq{t=119,9895\approx120}
        \end{minipage}
